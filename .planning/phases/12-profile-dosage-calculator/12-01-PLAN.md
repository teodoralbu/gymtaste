```typescript
import { describe, it, expect } from 'vitest'
import { calculateDosages, DOSAGE_RANGES, SAFETY_DISCLAIMER } from './dosage'
import type { FitnessGoal, DosageResult } from './dosage'

describe('calculateDosages', () => {
  const fullProduct = { caffeineMg: 200, citrullineG: 6, betaAlanineG: 3.2 }

  it('returns 3 results for product with all ingredients', () => {
    const results = calculateDosages(80, 'muscle_gain', fullProduct)
    expect(results).toHaveLength(3)
  })

  it('calculates caffeine range for muscle_gain (3-4 mg/kg)', () => {
    const results = calculateDosages(80, 'muscle_gain', fullProduct)
    const caffeine = results.find(r => r.ingredientName === 'Caffeine')!
    expect(caffeine.recommendedRange).toEqual({ min: 240, max: 320 })
    expect(caffeine.unit).toBe('mg')
    expect(caffeine.productAmountPerServing).toBe(200)
  })

  it('calculates caffeine range for fat_loss (4-6 mg/kg)', () => {
    const results = calculateDosages(80, 'fat_loss', fullProduct)
    const caffeine = results.find(r => r.ingredientName === 'Caffeine')!
    expect(caffeine.recommendedRange).toEqual({ min: 320, max: 480 })
  })

  it('calculates caffeine range for endurance (3-6 mg/kg)', () => {
    const results = calculateDosages(80, 'endurance', fullProduct)
    const caffeine = results.find(r => r.ingredientName === 'Caffeine')!
    expect(caffeine.recommendedRange).toEqual({ min: 240, max: 480 })
  })

  it('caps caffeine at 400mg absolute max', () => {
    const results = calculateDosages(100, 'fat_loss', fullProduct)
    const caffeine = results.find(r => r.ingredientName === 'Caffeine')!
    // 100kg * 6 = 600, but capped at 400
    expect(caffeine.recommendedRange.max).toBe(400)
  })

  it('beta-alanine range is flat (not weight-dependent)', () => {
    const r1 = calculateDosages(60, 'muscle_gain', fullProduct)
    const r2 = calculateDosages(100, 'fat_loss', fullProduct)
    const ba1 = r1.find(r => r.ingredientName === 'Beta-Alanine')!
    const ba2 = r2.find(r => r.ingredientName === 'Beta-Alanine')!
    expect(ba1.recommendedRange).toEqual({ min: 3.2, max: 6.4 })
    expect(ba2.recommendedRange).toEqual({ min: 3.2, max: 6.4 })
    expect(ba1.unit).toBe('g')
  })

  it('citrulline range is flat (not weight-dependent)', () => {
    const results = calculateDosages(80, 'endurance', fullProduct)
    const cit = results.find(r => r.ingredientName === 'Citrulline')!
    expect(cit.recommendedRange).toEqual({ min: 6, max: 8 })
    expect(cit.unit).toBe('g')
  })

  it('withinRange is true when product amount is in range', () => {
    const product = { caffeineMg: 300, citrullineG: 7, betaAlanineG: 4 }
    const results = calculateDosages(80, 'muscle_gain', product)
    const cit = results.find(r => r.ingredientName === 'Citrulline')!
    expect(cit.withinRange).toBe(true)
    const ba = results.find(r => r.ingredientName === 'Beta-Alanine')!
    expect(ba.withinRange).toBe(true)
  })

  it('withinRange is false when product amount is out of range', () => {
    const product = { caffeineMg: 500, citrullineG: 2, betaAlanineG: 8 }
    const results = calculateDosages(80, 'muscle_gain', product)
    const caffeine = results.find(r => r.ingredientName === 'Caffeine')!
    expect(caffeine.withinRange).toBe(false)
    const cit = results.find(r => r.ingredientName === 'Citrulline')!
    expect(cit.withinRange).toBe(false)
  })

  it('excludes null ingredients from results', () => {
    const results = calculateDosages(80, 'muscle_gain', { caffeineMg: 200, citrullineG: null, betaAlanineG: 3.2 })
    expect(results).toHaveLength(2)
    expect(results.find(r => r.ingredientName === 'Citrulline')).toBeUndefined()
  })

  it('returns empty array when all ingredients are null', () => {
    const results = calculateDosages(80, 'muscle_gain', { caffeineMg: null, citrullineG: null, betaAlanineG: null })
    expect(results).toHaveLength(0)
  })

  it('returns empty array for zero weight', () => {
    const results = calculateDosages(0, 'muscle_gain', fullProduct)
    expect(results).toHaveLength(0)
  })
})

describe('DOSAGE_RANGES', () => {
  it('has caffeine source referencing ISSN', () => {
    expect(DOSAGE_RANGES.caffeine.source).toContain('ISSN')
  })
  it('has beta_alanine source referencing ISSN', () => {
    expect(DOSAGE_RANGES.beta_alanine.source).toContain('ISSN')
  })
})

describe('SAFETY_DISCLAIMER', () => {
  it('contains ISSN reference', () => {
    expect(SAFETY_DISCLAIMER).toContain('ISSN')
  })
  it('contains medical advice warning', () => {
    expect(SAFETY_DISCLAIMER).toContain('not medical advice')
  })
})
```

**GREEN phase -- implement `src/lib/dosage.ts`:**

```typescript
export type FitnessGoal = 'muscle_gain' | 'fat_loss' | 'endurance'

export interface DosageResult {
  ingredientName: string
  productAmountPerServing: number
  unit: string
  recommendedRange: { min: number; max: number }
  withinRange: boolean
  note: string
}

export const DOSAGE_RANGES = {
  caffeine: {
    minPerKg: 3,
    maxPerKg: 6,
    absoluteMax: 400,
    unit: 'mg',
    source: 'ISSN Position Stand: Caffeine and Exercise Performance (2021)',
  },
  beta_alanine: {
    min: 3.2,
    max: 6.4,
    unit: 'g',
    source: 'ISSN Position Stand: Beta-Alanine (2015)',
  },
  citrulline: {
    min: 6,
    max: 8,
    unit: 'g',
    source: 'ISSN Position Stand: Pre-Exercise Nutrition',
  },
} as const

const GOAL_CAFFEINE_MODIFIER: Record<FitnessGoal, { min: number; max: number }> = {
  muscle_gain: { min: 3, max: 4 },
  fat_loss: { min: 4, max: 6 },
  endurance: { min: 3, max: 6 },
}

export { GOAL_CAFFEINE_MODIFIER }

export const SAFETY_DISCLAIMER =
  'These ranges are general guidance based on ISSN position stands and are not medical advice. Individual tolerance varies. Consult a healthcare professional before starting any supplement regimen. Do not exceed manufacturer-recommended serving sizes.'

export function calculateDosages(
  weightKg: number,
  goal: FitnessGoal,
  product: { caffeineMg: number | null; citrullineG: number | null; betaAlanineG: number | null }
): DosageResult[] {
  if (weightKg <= 0) return []

  const results: DosageResult[] = []

  // Caffeine (weight-dependent, goal-modified)
  if (product.caffeineMg != null) {
    const modifier = GOAL_CAFFEINE_MODIFIER[goal]
    const min = Math.round(modifier.min * weightKg)
    const max = Math.min(Math.round(modifier.max * weightKg), DOSAGE_RANGES.caffeine.absoluteMax)
    results.push({
      ingredientName: 'Caffeine',
      productAmountPerServing: product.caffeineMg,
      unit: DOSAGE_RANGES.caffeine.unit,
      recommendedRange: { min, max },
      withinRange: product.caffeineMg >= min && product.caffeineMg <= max,
      note: DOSAGE_RANGES.caffeine.source,
    })
  }

  // Citrulline (flat range)
  if (product.citrullineG != null) {
    results.push({
      ingredientName: 'Citrulline',
      productAmountPerServing: product.citrullineG,
      unit: DOSAGE_RANGES.citrulline.unit,
      recommendedRange: { min: DOSAGE_RANGES.citrulline.min, max: DOSAGE_RANGES.citrulline.max },
      withinRange: product.citrullineG >= DOSAGE_RANGES.citrulline.min && product.citrullineG <= DOSAGE_RANGES.citrulline.max,
      note: DOSAGE_RANGES.citrulline.source,
    })
  }

  // Beta-Alanine (flat range)
  if (product.betaAlanineG != null) {
    results.push({
      ingredientName: 'Beta-Alanine',
      productAmountPerServing: product.betaAlanineG,
      unit: DOSAGE_RANGES.beta_alanine.unit,
      recommendedRange: { min: DOSAGE_RANGES.beta_alanine.min, max: DOSAGE_RANGES.beta_alanine.max },
      withinRange: product.betaAlanineG >= DOSAGE_RANGES.beta_alanine.min && product.betaAlanineG <= DOSAGE_RANGES.beta_alanine.max,
      note: DOSAGE_RANGES.beta_alanine.source,
    })
  }

  return results
}
```
