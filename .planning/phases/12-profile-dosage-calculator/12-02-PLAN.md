**Create `src/app/products/[slug]/DosageCard.tsx`:**

```tsx
'use client'

import { useAuth } from '@/context/auth-context'
import { calculateDosages, SAFETY_DISCLAIMER } from '@/lib/dosage'
import type { FitnessGoal } from '@/lib/dosage'
import Link from 'next/link'

interface DosageCardProps {
  caffeineMg: number | null
  citrullineG: number | null
  betaAlanineG: number | null
}

export function DosageCard({ caffeineMg, citrullineG, betaAlanineG }: DosageCardProps) {
  const { profile, loading } = useAuth()

  // Hide entirely if no ingredient data at all
  if (caffeineMg == null && citrullineG == null && betaAlanineG == null) {
    return null
  }

  // Loading state
  if (loading) return null

  // No profile or no weight -- show CTA
  if (!profile?.weight_kg) {
    return (
      <div className="card" style={{ padding: '20px', marginBottom: '40px' }}>
        <h2 style={{
          fontSize: '16px',
          fontWeight: 800,
          marginBottom: '8px',
          color: 'var(--text)',
          letterSpacing: '-0.01em',
        }}>
          Dosage Calculator
        </h2>
        <p style={{ fontSize: '14px', color: 'var(--text-muted)', marginBottom: '12px' }}>
          Add your body stats in settings to get personalized dosage recommendations.
        </p>
        <Link
          href="/settings"
          style={{
            display: 'inline-block',
            fontSize: '14px',
            fontWeight: 600,
            color: 'var(--accent)',
            textDecoration: 'none',
          }}
        >
          Go to Settings
        </Link>
      </div>
    )
  }

  const goal: FitnessGoal = profile.fitness_goal ?? 'muscle_gain'
  const results = calculateDosages(
    profile.weight_kg,
    goal,
    { caffeineMg, citrullineG, betaAlanineG }
  )

  if (results.length === 0) return null

  return (
    <div className="card" style={{ padding: '20px', marginBottom: '40px' }}>
      <h2 style={{
        fontSize: '16px',
        fontWeight: 800,
        marginBottom: '4px',
        color: 'var(--text)',
        letterSpacing: '-0.01em',
      }}>
        Dosage Calculator
      </h2>
      <p style={{
        fontSize: '12px',
        color: 'var(--text-dim)',
        marginBottom: '16px',
      }}>
        Based on your weight ({profile.weight_kg}kg) and goal ({goal.replace(/_/g, ' ')})
      </p>

      <div style={{ display: 'flex', flexDirection: 'column', gap: '12px' }}>
        {results.map((result) => (
          <div
            key={result.ingredientName}
            style={{
              display: 'flex',
              justifyContent: 'space-between',
              alignItems: 'center',
              padding: '12px 16px',
              borderRadius: 'var(--radius-lg)',
              backgroundColor: 'var(--bg-elevated)',
              border: '1px solid var(--border)',
            }}
          >
            <div>
              <div style={{
                fontSize: '14px',
                fontWeight: 700,
                color: 'var(--text)',
                marginBottom: '2px',
              }}>
                {result.ingredientName}
              </div>
              <div style={{
                fontSize: '12px',
                color: 'var(--text-dim)',
              }}>
                Product: {result.productAmountPerServing}{result.unit}/serving
              </div>
            </div>
            <div style={{ textAlign: 'right' }}>
              <div style={{
                fontSize: '14px',
                fontWeight: 700,
                color: result.withinRange ? '#00E676' : '#FF9100',
              }}>
                {result.withinRange ? 'In range' : 'Outside range'}
              </div>
              <div style={{
                fontSize: '12px',
                color: 'var(--text-dim)',
              }}>
                {result.recommendedRange.min}-{result.recommendedRange.max}{result.unit}
              </div>
            </div>
          </div>
        ))}
      </div>

      {/* Safety disclaimer -- non-dismissable */}
      <div style={{
        backgroundColor: 'var(--bg-elevated)',
        border: '1px solid var(--border)',
        borderRadius: 'var(--radius-lg)',
        padding: '12px 16px',
        marginTop: '16px',
      }}>
        <p style={{ fontSize: '12px', color: 'var(--text-dim)', lineHeight: 1.5, margin: 0 }}>
          <strong style={{ color: 'var(--text-muted)' }}>Disclaimer:</strong>{' '}
          {SAFETY_DISCLAIMER}
        </p>
      </div>
    </div>
  )
}
```
